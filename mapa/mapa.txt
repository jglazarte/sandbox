<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>

<script>
    /* --- 2. CONFIGURACIÓN Y VARIABLES GLOBALES --- */
    
    const CONFIG_CATEGORIAS = {
        "Lugares":  { color: "#d35400", hex: "#d35400" }, // Naranja
        "Personas": { color: "#2980b9", hex: "#2980b9" }, // Azul
        "Sucesos":  { color: "#c0392b", hex: "#c0392b" }  // Rojo
    };

    const RUTA_IMAGENES = "img/";

    var map = L.map('map', {
        center: [-36.7782, -59.8514],
        zoom: 14,
        minZoom: 13,
        maxBounds: L.latLngBounds([-36.8500, -59.9500], [-36.7000, -59.7500]),
        maxBoundsViscosity: 1.0,
        zoomControl: false
    });
    L.control.zoom({ position: 'topright' }).addTo(map);

    L.tileLayer('https://{s}.basemaps.cartocdn.com/rastertiles/voyager/{z}/{x}/{y}{r}.png', {
        attribution: '&copy; OpenStreetMap &copy; CARTO',
        subdomains: 'abcd',
        maxZoom: 20
    }).addTo(map);

    var marcadoresLayer = L.layerGroup().addTo(map);
    var listaContainer = document.getElementById('lista-lugares');

    /* --- 3. BASE DE DATOS (ARRAY) --- */
    const datosLugares = [
        {
            id: 1,
            titulo: "Teatro Español",
            categoria: "Lugares",
            lat: -36.7794,
            lng: -59.8584,
            img: "teatro.jpg", 
            desc: "Inaugurado en 1897, es una joya arquitectónica y centro cultural de la ciudad."
        },
        {
            id: 2,
            titulo: "Cementerio Municipal",
            categoria: "Lugares",
            lat: -36.7645,
            lng: -59.8465,
            img: "cementerio.jpg",
            desc: "Destacado por su obra de Salamone: el imponente ángel exterminador Art Decó."
        },
        {
            id: 3,
            titulo: "Bautismo de la Libertad",
            categoria: "Sucesos",
            lat: -36.7800,
            lng: -59.8550,
            img: "libertad.jpg",
            desc: "Histórico suceso militar donde se bautizaron las banderas de los regimientos."
        },
        {
            id: 4,
            titulo: "Monumento a Cervantes",
            categoria: "Personas",
            lat: -36.7850, 
            lng: -59.8500,
            img: "cervantes.jpg",
            desc: "Homenaje al autor del Quijote, celebrando el título de Ciudad Cervantina."
        },
        {
            id: 5,
            titulo: "Plaza Gral. José de San Martín",
            categoria: "Lugares",
            lat: -36.7775, 
            lng: -59.8636,
            // Aquí usamos un array en lugar de un string para múltiples fotos
            imgs: ["fuertessm.jpg", "plazasmner.jpg", "plazasmner2.jpg", "plazasm.jpg"], 
            desc: "Lugar de asentamiento del Mojón de Rosas, se emplazó el Fuerte San Serapio Mártir del Azul. A principios del S. XX el paisajista francés Carlos Thays realizó la obra en la plaza con una fuente con mujeres desnudas en el centro (las llamaban las nereidas). En 1912 500 vecinos se opusieron a mantener esas esculturas por lo que fue demolida. Años más tarde Salamone le dio su aspecto actual"
        }
    ];

    /* --- 4. MOTOR (LÓGICA QUE GENERA EL MAPA) --- */
    
    function crearIcono(color) {
        return L.divIcon({
            className: 'custom-div-icon',
            html: `<div class="custom-pin" style="background-color: ${color};"></div>`,
            iconSize: [30, 30],
            iconAnchor: [15, 30],
            popupAnchor: [0, -30]
        });
    }

    function iniciarApp() {
        datosLugares.forEach(lugar => {
            const config = CONFIG_CATEGORIAS[lugar.categoria] || { color: '#7f8c8d' };
            const color = config.color;

            const marcador = L.marker([lugar.lat, lugar.lng], {icon: crearIcono(color)});
            
            // --- LÓGICA IMAGEN vs CARRUSEL ---
            let contenidoMedia = "";
            
            // Si tiene el campo 'imgs' (es un array), crea carrusel
            if (lugar.imgs && lugar.imgs.length > 0) {
                // Generar HTML del carrusel
                let slidesHtml = lugar.imgs.map(img => 
                    `<div class="swiper-slide"><img src="${RUTA_IMAGENES}${img}" onerror="this.src='https://via.placeholder.com/300x150?text=Error'"></div>`
                ).join('');
                
                // ID único para este swiper
                const swiperId = `swiper-${lugar.id}`;
                
                contenidoMedia = `
                    <div class="swiper ${swiperId}">
                        <div class="swiper-wrapper">
                            ${slidesHtml}
                        </div>
                        <div class="swiper-button-next"></div>
                        <div class="swiper-button-prev"></div>
                    </div>
                `;
            } 
            // Si solo tiene 'img' (un string), usa imagen simple
            else if (lugar.img) {
                contenidoMedia = `<img src="${RUTA_IMAGENES}${lugar.img}" class="popup-img" onerror="this.src='https://via.placeholder.com/300x150?text=Sin+Foto'">`;
            }

            const popupContent = `
                <div class="popup-content">
                    <div class="popup-header">
                        <h3 style="border-color: ${color}">${lugar.titulo}</h3>
                    </div>
                    ${contenidoMedia}
                    <div class="popup-body">
                        ${lugar.desc}
                    </div>
                    <div class="popup-footer">
                        <span style="font-size:10px; color:#999; text-transform:uppercase;">${lugar.categoria}</span>
                    </div>
                </div>
            `;
            
            marcador.bindPopup(popupContent);

            // IMPORTANTE: Inicializar Swiper cuando el popup se abra
            marcador.on('popupopen', function() {
                if (lugar.imgs) {
                    const selector = `.swiper-${lugar.id}`;
                    // Buscar el swiper dentro del popup
                    const popupElement = document.querySelector(selector);
                    if(popupElement && !popupElement.classList.contains('swiper-initialized')) {
                        new Swiper(selector, {
                            navigation: {
                                nextEl: '.swiper-button-next',
                                prevEl: '.swiper-button-prev',
                            },
                            loop: true // Permite hacer loop infinito de fotos
                        });
                    }
                }
            });

            lugar.markerRef = marcador; 
            marcadoresLayer.addLayer(marcador);

            // --- CREAR LISTA ---
            const itemLista = document.createElement('div');
            itemLista.className = 'place-item';
            itemLista.dataset.categoria = lugar.categoria;
            itemLista.style.borderLeftColor = color;
            itemLista.innerHTML = `
                <div class="place-number" style="color:${color}">${lugar.id}</div>
                <div class="place-info">
                    <h4>${lugar.titulo}</h4>
                    <span style="color:${color}">${lugar.categoria}</span>
                </div>
            `;
            itemLista.addEventListener('click', () => {
                map.flyTo([lugar.lat, lugar.lng], 16);
                setTimeout(() => marcador.openPopup(), 1000);
            });
            listaContainer.appendChild(itemLista);
        });
    }

    /* --- 5. LÓGICA DE FILTROS --- */
    function filtrar(categoria, btnElement) {
        document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
        btnElement.classList.add('active');

        const items = document.querySelectorAll('.place-item');
        items.forEach(item => {
            if (categoria === 'todos' || item.dataset.categoria === categoria) {
                item.style.display = 'flex';
            } else {
                item.style.display = 'none';
            }
        });

        marcadoresLayer.clearLayers();
        datosLugares.forEach(lugar => {
            if (categoria === 'todos' || lugar.categoria === categoria) {
                marcadoresLayer.addLayer(lugar.markerRef);
            }
        });
    }

    iniciarApp();

</script>
